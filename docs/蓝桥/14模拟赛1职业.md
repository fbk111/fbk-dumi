> 数组有reverse的方法，想啥呢
```js
function isPalindromeStr(str) {
  // 在这里写入具体的实现逻辑
  // 返回值是 boolean 类型， 如果是回文字符串应该返回 true, 否则返回 false
  if(typeof str !='string') return false
  else{
    return str.split('').reverse().join('')==str
  }
};
module.exports = isPalindromeStr; // 检测需要请勿删除
```
