## reverse底层原理

        Array.prototype.myReverse = function () {
            var len = this.length;
            for (var i = 0; i < len; i++) {
                var temp = this[i];
                this[i] = this[len - 1 - i];
                this[len - 1 - i] = temp;
            }
            return this;
        }

## 防抖
```js
  function debounce(handle,delay){
         let timer=null
         return function(){
            let _self=this,
            _args=arguments
            clearTimeout(timer)
            timer=setTimeout(() => {
                handle.apply(_self,_args)
            }, delay);
         }
      }
```

      function throttle(handle,wait){
        let lastTime=0
        return function(e){
            let nowTime=new Date().getTime()
            if(nowTime-lastTime>wait){
                handle.apply(this,arguments)
                lastTime=nowTime
            }
        }
      }
